#### Day 5: Order ID Generator Service

**Critical Implementation:**
```javascript
// services/orderIdGenerator.js
export async function generateOrderId() {
  const today = new Date();
  const DD = String(today.getDate()).padStart(2, '0');
  const MM = String(today.getMonth() + 1).padStart(2, '0');
  const YY = String(today.getFullYear()).slice(-2);
  const dateKey = `${today.getFullYear()}-${MM}-${DD}`;
  
  // Atomic increment
  const counter = await Counter.findOneAndUpdate(
    { _id: dateKey },
    { $inc: { seq: 1 } },
    { upsert: true, new: true, setDefaultsOnInsert: true }
  );
  
  const SS = String(counter.seq).padStart(2, '0');
  return `${DD}${MM}${YY}${SS}`;
}
```

**Testing:**

- âœ… Verify format: DDMMYYSS

// middleware/metrics.js
import logger from '../config/logger.js';

const metricsStore = {
  requests: 0,
  errors: 0,
  responseTime: []
};

export const metricsMiddleware = (req, res, next) => {
  const start = Date.now();
  
  res.on('finish', () => {
    const duration = Date.now() - start;
    metricsStore.requests++;
    metricsStore.responseTime.push(duration);
    
    if (res.statusCode >= 500) {
      metricsStore.errors++;
    }
    
    logger.debug('Request metrics', {
      method: req.method,
      url: req.url,
      status: res.statusCode,
      duration: `${duration}ms`
    });
  });
  
  next();
};

// Metrics endpoint (admin only)
export const getMetrics = (req, res) => {
  const avgResponseTime = metricsStore.responseTime.reduce((a, b) => a + b, 0) 
    / metricsStore.responseTime.length;
  
  res.json({
    totalRequests: metricsStore.requests,
    errorCount: metricsStore.errors,
    avgResponseTime: avgResponseTime.toFixed(2),
    uptime: process.uptime()
  });
};

### Common Issues & Solutions

**1. Order ID Collision**
```javascript
// Symptom: Duplicate order IDs
// Solution: Ensure atomic counter with proper locking

const generateOrderId = async () => {
  const session = await mongoose.startSession();
  session.startTransaction();
  
  try {
    const counter = await Counter.findOneAndUpdate(
      { _id: dateKey },
      { $inc: { seq: 1 } },
      { upsert: true, new: true, session }
    );
    
    await session.commitTransaction();
    return formatOrderId(counter.seq);
  } catch (error) {
    await session.abortTransaction();
    throw error;
  } finally {
    session.endSession();
  }
};
```

**3. MongoDB Connection Issues**
```javascript
// Add retry logic
const connectDB = async (retries = 5) => {
  for (let i = 0; i < retries; i++) {
    try {
      await mongoose.connect(process.env.MONGODB_URI);
      logger.info('MongoDB connected');
      return;
    } catch (error) {
      logger.error(`Connection attempt ${i + 1} failed`);
      if (i === retries - 1) throw error;
      await new Promise(resolve => setTimeout(resolve, 5000));
    }
  }
};
```

// Implement refresh token mechanism
export const refreshToken = async (req, res) => {
  const refreshToken = req.cookies.refreshToken;
  
  if (!refreshToken) {

  }
  
  try {
    const decoded = jwt.verify(refreshToken, process.env.REFRESH_SECRET);
    const user = await User.findById(decoded.id);

    res.cookie('token', newAccessToken, { httpOnly: true });
    res.json({ success: true });
  } catch (error) {

  }
